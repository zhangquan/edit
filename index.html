(function () {
    "use strict";

    var $, editor, tokens, showError;

    function showValue() {
        $(".log").html(editor.doc.getValue());
    }

    showError = function (msg) {
        $(".log").html($(".log").html() + msg + "d");
    };

    function showParse() {
        var t = tokens(editor.doc.getValue());
        $(".log").html(JSON.stringify(t));
    }
    
}());
